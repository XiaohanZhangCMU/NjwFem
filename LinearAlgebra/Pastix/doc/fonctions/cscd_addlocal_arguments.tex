  \argdesc{ia          }{Column starting positions in first CSCD matrix.}
  \argdesc{ja          }{Rows in first CSCD matrix.}
  \argdesc{a           }{Values in first CSCD matrix (can be NULL).}
  \argdesc{l2g         }{Global column number map for first CSCD matrix.}
  \argdesc{addn        }{Size of the second CSCD matrix (to be added to base).}
  \argdesc{addia       }{Column starting positions in second CSCD matrix.}
  \argdesc{addja       }{Rows in second CSCD matrix.}
  \argdesc{adda        }{Values in second CSCD (can be NULL $\rightarrow$ add \o).}
  \argdesc{addl2g      }{Global column number map for second CSCD matrix.}
  \argdesc{newn        }{Size of output CSCD matrix (same as n).}
  \argdesc{newia       }{Column starting positions in output CSCD matrix.}
  \argdesc{newja       }{Rows in output CSCD matrix.}
  \argdesc{newa        }{Values in outpur CSCD matrix.}
  \argdesc{malloc\_flag}{Flag: Function call is internal to \pastix{}.}
