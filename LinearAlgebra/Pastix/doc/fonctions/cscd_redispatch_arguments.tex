\begin{arguments}
  \argdesc{n   }{Number of local columns}
  \argdesc{ia  }{First cscd starting index of each column in \texttt{ja} and \texttt{a}}
  \argdesc{ja  }{Row of each element in first CSCD}
  \argdesc{a   }{Value of each CSCD in first CSCD (can be NULL)}
  \argdesc{rhs }{Right-hand-side member corresponding to the first CSCD (can be NULL)}
  \argdesc{l2g }{Local to global column numbers for first CSCD}
  \argdesc{dn  }{Number of local columns}
  \argdesc{dia }{New CSCD starting index of each column in \texttt{ja} and \texttt{a}}
  \argdesc{dja }{Row of each element in new CSCD}
  \argdesc{da  }{Value of each CSCD in new CSCD}
  \argdesc{rhs }{Right-hand-side member corresponding to the new CSCD}
  \argdesc{dl2g}{Local to global column numbers for new CSCD}
  \argdesc{comm}{MPI communicator}
\end{arguments}
